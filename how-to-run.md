# audiotocpp
Convert MP3/WAV/OGG/FLAC/M4A/BANK/BNK to C++ Code (.h)
Launch:
mp3 -> h : audiotoh music.mp3 music.h
wav -> h : audiotoh sound.wav sound_data.h mySound
ogg -> h : audiotoh track.ogg track.h
flac -> h : audiotoh effect.flac effect_data.h myFlac
m4a -> h : audiotoh voice.m4a voice_data.h myVoice
bank -> h : audiotoh bank.bank bank_data.h bankAudio
bnk -> h : audiotoh file.bnk file_data.h fileBank
